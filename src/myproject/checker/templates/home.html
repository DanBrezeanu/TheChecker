The solution for every problem must be written into the specified file. The input (according to the problem's text) must be read from the specified input file. 
 
#include <fstream>

int main(int argc, char* argv[]) {
   ifstream in("x.in");    // read from file x.in
   ofstream out("x.out");  // write to file x.out

   in >> a;
   out << a;

   return 0;
}
                                                                                        
                                        
 
#include <stdio.h>

int main(int argc, char* argv[]) {
   FILE *fi = fopen("x.in", "r");   // read from file x.in
   FILE *fo = fopen("x.out", "w");  // write to file x.out

   fscanf(fi, "%d", &a);
   fprintf(fo, "%d\n", a);

   return 0;
}
